Demo

Examples of desoSlide jQuery plugin

For installation instructions, refer to the documentation.

1 Default options, with caption

Result

Code

$('#demo1_thumbs').desoSlide({
    main: {
        container: '#demo1_main_image',
        cssClass: 'img-responsive'
    },
    caption: true
});

2 Autostarting, displaying the second image at loading

No link and no keys. Just standard controls.

Result

Code

$('#demo2_thumbs').desoSlide({
    main: {
        container: '#demo2_main_image',
        cssClass: 'img-responsive'
    },
    auto: {
        start: true
    },
    first: 1,
    effect: 'sideFade',
    controls: {
        keys: false
    }
});

3 Displaying the overlay by hovering the main image

No link provided and no controls.

Result

Code

$('#demo3_thumbs').desoSlide({
    main: {
        container: '#demo3_main_image',
        cssClass: 'img-responsive'
    },
    effect: 'flip',
    overlay: 'hover',
    caption: true,
    controls: {
        enable: false,
        keys: false
    }
});

4 Autoloading, hiding the overlay

Result

Code

$('#demo4_thumbs').desoSlide({
    main: {
        container: '#demo4_main_image',
        cssClass: 'img-responsive'
    },
    auto: {
        start: true
    },
    effect: 'rotate',
    overlay: 'none',
    controls: {
        enable: false,
        keys: false
    }
});
